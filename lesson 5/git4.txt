# 1. склонировать репозиторий в текущую папку (на это указывает точка в конце)
git clone git@github.com:TOP-Python319/IamchukMO.git .

# 2. Добавил файлы домашних заданий

# 3. Пометил файлы к изменениям (все файлы из текущего каталога)
git add .

# 4. Закоммитили изменения
git commit -m "Я сделал первое ДЗ"

# 5. Отправили изменения, но ветки master не существует на удалённом репозитории
git push

# 6. Отправили изменения вместе с новой веткой
git push --set-upstream origin master

# 7. На github мы получили уведомление о том что есть изменения с другой ветки и их надо слить
Compare & Pull request

# 8. Мы приняли все изменения
Merge

# 9. УР - ветка main получила все изменения (файлы) из ветки master, ЛР не имеет этих изменений, надо их получить

# 10. Переключаемся на ветку main
git checkout main

# 11. Получаем все изменения в ветке main из УР в ЛР
git pull
