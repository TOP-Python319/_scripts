#1. Проверяем ветку по умолчанию (master или main)
git config --list --show-origin

#2. Меняем ветку по умолчанию на main
git config --global init.defaultBranch main

#3. Добавил файлы домашних заданий

#4. Пометил файлы к изменениям (все файлы из текущего каталога)
git add .

#5. Закоммитили изменения
git commit -m "Я сделал первое ДЗ"

#6. Отправили изменения
git push

###############
git checkout -b main  # если у вас в ЛР текущая ветка master, то с помощью checkout -b можно создать новую ветку и переключиться на неё

###############
git branch  # узнать текущую ветку, помечена звёздочкой
