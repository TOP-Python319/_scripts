# для установки
pip install virtualenv

# посмотреть список пакетов
pip list

# есть папка с проектом и в её корне создаётся виртуальное окружение


# чтобы создать новое виртуальное окружение:
# python3 -m {command} {name}
# например
# python3 -m venv venv

# чтобы переключиться на новое виртуальное окружение  нужно набрать команду в той же папке где лежит виртуальное окружение:
# Linux/Mac
# source venv/bin/activate
# Windows
# venv\Scripts\activate


# чтобы деактивировать виртуальное окружение
# Linux/Mac/Windows:
# deactivate


# для того чтобы зафиксировать версии можно сохранить их в txt файле
# например requirements.txt
# для того чтобы установить из файла пакеты:
# pip install -r requirements.txt 


# узнать какое окружение python запущено:
# pip -V


# в каждой копии терминала можно запустить своё виртуальное окружение python
